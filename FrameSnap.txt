    <script>
        let extractedFrames = [];
        let isProcessing = false;

        const videoInput = document.getElementById('videoInput');
        const uploadSection = document.getElementById('uploadSection');
        const editorSection = document.getElementById('editorSection');
        const sourceVideo = document.getElementById('sourceVideo');
        const extractBtn = document.getElementById('extractBtn');
        const gallery = document.getElementById('gallery');
        const galleryContainer = document.getElementById('galleryContainer');
        const emptyGallery = document.getElementById('emptyGallery');
        const progressBar = document.getElementById('progressBar');
        const progressContainer = document.getElementById('progressContainer');
        const percentText = document.getElementById('percentText');
        const frameCounter = document.getElementById('frameCounter');
        const downloadAllBtn = document.getElementById('downloadAllBtn');
        const rangeControls = document.getElementById('rangeControls');
        const rangeStart = document.getElementById('rangeStart');
        const rangeEnd = document.getElementById('rangeEnd');
        const downloadZipBtn = document.getElementById('downloadZipBtn');
        const resetBtn = document.getElementById('resetBtn');
        const modal = document.getElementById('modal');
        const modalImg = document.getElementById('modalImg');
        const timeStep = document.getElementById('timeStep');
        const intervalValue = document.getElementById('intervalValue');

        lucide.createIcons();

        // Update interval text
        timeStep.addEventListener('input', (e) => {
            intervalValue.innerText = `${parseFloat(e.target.value).toFixed(2)}s`;
        });

        function showToast(msg, iconName = 'check-circle') {
            const toast = document.getElementById('toast');
            const toastMsg = document.getElementById('toastMsg');
            const toastIcon = document.getElementById('toastIcon');
            toastMsg.innerText = msg;
            toastIcon.innerHTML = `<i data-lucide="${iconName}" class="w-5 h-5"></i>`;
            lucide.createIcons({ attrs: { 'data-lucide': iconName } });
            toast.classList.remove('translate-y-24', 'opacity-0');
            setTimeout(() => toast.classList.add('translate-y-24', 'opacity-0'), 4000);
        }

        videoInput.addEventListener('change', (e) => {
            const file = e.target.files[0];
            if (!file) return;
            sourceVideo.src = URL.createObjectURL(file);
            uploadSection.classList.add('hidden');
            editorSection.classList.remove('hidden');
            sourceVideo.onloadedmetadata = () => showToast("Video loaded successfully.");
        });

        resetBtn.addEventListener('click', () => {
            if (isProcessing && !confirm("Stop extraction and change video?")) return;
            location.reload();
        });

        // Simple pixel difference scene detection
        function getFrameDiff(ctx1, ctx2, width, height) {
            const data1 = ctx1.getImageData(0, 0, width, height).data;
            const data2 = ctx2.getImageData(0, 0, width, height).data;
            let diff = 0;
            // Sample pixels for performance (every 10th pixel)
            for (let i = 0; i < data1.length; i += 40) {
                diff += Math.abs(data1[i] - data2[i]);
                diff += Math.abs(data1[i+1] - data2[i+1]);
                diff += Math.abs(data1[i+2] - data2[i+2]);
            }
            return diff / (width * height);
        }

        extractBtn.addEventListener('click', async () => {
            if (isProcessing) return;
            
            const extractMode = document.querySelector('input[name="extractMode"]:checked').value;
            const video = sourceVideo;
            const duration = video.duration;
            
            isProcessing = true;
            extractBtn.disabled = true;
            extractBtn.classList.add('opacity-50', 'grayscale');
            progressContainer.classList.remove('hidden');
            
            if (emptyGallery) emptyGallery.remove();
            extractedFrames = [];
            gallery.innerHTML = '';
            
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            const prevCanvas = document.createElement('canvas');
            const prevCtx = prevCanvas.getContext('2d');
            
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;
            prevCanvas.width = video.videoWidth;
            prevCanvas.height = video.videoHeight;

            let currentTime = 0;
            let step = parseFloat(timeStep.value);
            let count = 0;
            const autoThreshold = 1.5; // Visual difference sensitivity

            while (currentTime <= duration) {
                video.currentTime = currentTime;
                await new Promise(resolve => {
                    const onSeeked = () => {
                        video.removeEventListener('seeked', onSeeked);
                        resolve();
                    };
                    video.addEventListener('seeked', onSeeked);
                });

                ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
                
                let shouldSnap = true;
                if (extractMode === 'auto' && count > 0) {
                    const diff = getFrameDiff(ctx, prevCtx, canvas.width, canvas.height);
                    shouldSnap = diff > autoThreshold;
                }

                if (shouldSnap) {
                    const dataUrl = canvas.toDataURL('image/jpeg', 0.85);
                    const blob = await (await fetch(dataUrl)).blob();
                    const frameData = {
                        id: ++count,
                        dataUrl: dataUrl,
                        blob: blob,
                        timestamp: currentTime.toFixed(2)
                    };
                    extractedFrames.push(frameData);
                    addFrameToGallery(frameData);
                    
                    // Update prev canvas only when we snap
                    prevCtx.drawImage(canvas, 0, 0);
                }

                // If auto, check more frequently but only snap on changes
                currentTime += (extractMode === 'auto') ? 0.2 : step;
                
                const progress = Math.min(100, (currentTime / duration) * 100);
                progressBar.style.width = `${progress}%`;
                percentText.innerText = `${Math.round(progress)}%`;
                frameCounter.innerText = `${count} frames found`;
                if (count % 5 === 0) galleryContainer.scrollTop = galleryContainer.scrollHeight;
            }

            isProcessing = false;
            extractBtn.disabled = false;
            extractBtn.classList.remove('opacity-50', 'grayscale');
            showToast(`Extraction complete! ${count} frames.`, 'check-circle');
            downloadAllBtn.classList.remove('hidden');
            rangeControls.classList.remove('hidden');
            rangeEnd.value = count;
            lucide.createIcons();
        });

        function addFrameToGallery(frame) {
            const card = document.createElement('div');
            card.className = "frame-card bg-white border border-slate-200 rounded-2xl overflow-hidden shadow-sm hover:shadow-2xl animate-in fade-in slide-in-from-bottom-4 duration-500";
            card.innerHTML = `
                <div class="aspect-ratio-box group cursor-pointer">
                    <img src="${frame.dataUrl}" class="absolute inset-0 w-full h-full object-cover transition-transform group-hover:scale-105" loading="lazy">
                    <div class="absolute inset-0 bg-slate-900/40 opacity-0 group-hover:opacity-100 transition-all flex items-center justify-center">
                        <div class="bg-white/20 backdrop-blur-xl text-white p-4 rounded-full border border-white/30 transform scale-90 group-hover:scale-100 transition-all duration-300">
                            <i data-lucide="eye" class="w-6 h-6"></i>
                        </div>
                    </div>
                </div>
                <div class="p-5 flex items-center justify-between bg-white">
                    <div class="min-w-0">
                        <p class="text-[10px] font-black text-slate-300 uppercase tracking-widest mb-0.5">F#${frame.id}</p>
                        <p class="text-sm font-bold text-slate-700 truncate">${frame.timestamp}s</p>
                    </div>
                    <button class="download-btn bg-blue-50 p-3 rounded-xl text-blue-600 hover:text-white hover:bg-blue-600 transition-all shadow-sm active:scale-90">
                        <i data-lucide="download" class="w-5 h-5"></i>
                    </button>
                </div>
            `;
            card.querySelector('.aspect-ratio-box').onclick = () => {
                modalImg.src = frame.dataUrl;
                modal.classList.remove('hidden');
            };
            card.querySelector('.download-btn').onclick = (e) => {
                e.stopPropagation();
                const link = document.createElement('a');
                link.href = frame.dataUrl;
                link.download = `frame_${frame.id.toString().padStart(5, '0')}.jpg`;
                link.click();
            };
            gallery.appendChild(card);
            lucide.createIcons({ attrs: { 'data-lucide': null } });
        }

        document.getElementById('closeModal').onclick = () => modal.classList.add('hidden');
        modal.onclick = (e) => { if(e.target === modal) modal.classList.add('hidden') };

        async function createZip(framesToZip) {
            if (framesToZip.length === 0) return;
            const zip = new JSZip();
            const folder = zip.folder("video_frames");
            showToast(`Compressing ${framesToZip.length} files...`, 'package');
            framesToZip.forEach(frame => {
                const filename = `frame_${frame.id.toString().padStart(5, '0')}.jpg`;
                folder.file(filename, frame.blob);
            });
            const content = await zip.generateAsync({ type: "blob" });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(content);
            link.download = `frames_${new Date().getTime()}.zip`;
            link.click();
            showToast("Download started!");
        }

        downloadAllBtn.onclick = () => createZip(extractedFrames);
        downloadZipBtn.onclick = () => {
            const start = parseInt(rangeStart.value);
            const end = parseInt(rangeEnd.value);
            if (isNaN(start) || isNaN(end) || start < 1 || end > extractedFrames.length || start > end) {
                showToast("Invalid range selected", "alert-circle");
                return;
            }
            createZip(extractedFrames.filter(f => f.id >= start && f.id <= end));
        };
    </script>
