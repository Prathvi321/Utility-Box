    <script>
        // Data Structures
        let items = [
            { id: Date.now(), desc: 'Sample Product', qty: 1, rate: 1000, gst: 18 }

        function toggleSettings() {
            const panel = document.getElementById('settingsPanel');
            panel.classList.toggle('hidden');
        }

        function syncText(sourceId, targetId) {
            const val = document.getElementById(sourceId).value;
            if(targetId === 'displayCustName') {
                document.getElementById(targetId).innerText = val || 'Walking Customer';
            } else if(targetId === 'displayCustAddr') {
                document.getElementById(targetId).innerText = val || 'Address Details';
            } else {
                document.getElementById(targetId).innerText = val;
            }
        }

        // Store Logic (LocalStorage)
        function updateStore() {
            const data = {
                name: document.getElementById('storeName').value,
                gstin: document.getElementById('storeGstin').value,
                address: document.getElementById('storeAddress').value,
                contact: document.getElementById('storeContact').value
            };
            localStorage.setItem('gst_invoice_settings', JSON.stringify(data));
            applyStoreData(data);
        }

        function loadStoreData() {
            const raw = localStorage.getItem('gst_invoice_settings');
            if (raw) {
                const data = JSON.parse(raw);
                document.getElementById('storeName').value = data.name || '';
                document.getElementById('storeGstin').value = data.gstin || '';
                document.getElementById('storeAddress').value = data.address || '';
                document.getElementById('storeContact').value = data.contact || '';
                applyStoreData(data);
            }
        }

        function applyStoreData(data) {
            document.getElementById('displayStoreName').innerText = data.name || 'YOUR STORE';
            document.getElementById('displayStoreGstin').innerText = data.gstin || 'NOT ENTERED';
            document.getElementById('displayStoreAddress').innerText = data.address || 'Business address not provided.';
            document.getElementById('displayStoreContact').innerText = data.contact ? `Contact: ${data.contact}` : '';
        }

        // Optimized Item Logic to prevent focus loss
        function addRow() {
            items.push({ id: Date.now(), desc: '', qty: 1, rate: 0, gst: 18 });
            renderItems();
        }

        function removeItem(id) {
            if(items.length <= 1) return;
            items = items.filter(i => i.id !== id);
            renderItems();
        }

        // This function handles the focus-friendly updates
        function updateItem(id, field, value) {
            const item = items.find(i => i.id === id);
            if (field === 'desc') item[field] = value;
            else item[field] = parseFloat(value) || 0;
            
            // Instead of re-rendering everything, just update the calculations
            updateCalculations();
            
            // Also update the specific row amount display
            const rowTotal = item.qty * item.rate;
            const rowTax = (rowTotal * item.gst) / 100;
            const displayEl = document.getElementById(`row-amount-${id}`);
            if(displayEl) {
                displayEl.innerText = `₹${(rowTotal + rowTax).toLocaleString('en-IN', {minimumFractionDigits: 2})}`;
            }
        }

        function updateCalculations() {
            let subTotal = 0;
            let totalTax = 0;

            items.forEach(item => {
                const rowTotal = item.qty * item.rate;
                const rowTax = (rowTotal * item.gst) / 100;
                subTotal += rowTotal;
                totalTax += rowTax;
            });

            const grandTotal = subTotal + totalTax;

            document.getElementById('subTotal').innerText = `₹ ${subTotal.toLocaleString('en-IN', {minimumFractionDigits: 2})}`;
            document.getElementById('totalTax').innerText = `₹ ${totalTax.toLocaleString('en-IN', {minimumFractionDigits: 2})}`;
            document.getElementById('grandTotal').innerText = `₹ ${grandTotal.toLocaleString('en-IN', {minimumFractionDigits: 2})}`;
            document.getElementById('grandTotalWords').innerText = numberToWords(Math.round(grandTotal)) + " Rupees Only";
        }

        function renderItems() {
            const container = document.getElementById('itemRows');
            container.innerHTML = '';

            items.forEach(item => {
                const rowTotal = item.qty * item.rate;
                const rowTax = (rowTotal * item.gst) / 100;
                const rowGrand = rowTotal + rowTax;

                const tr = document.createElement('tr');
                tr.className = 'border-b border-slate-100 group';
                tr.innerHTML = `
                    <td class="py-4">

                    </td>
                    <td class="py-4 text-center">
                        <input type="number" value="${item.qty}" oninput="updateItem(${item.id}, 'qty', this.value)" class="w-16 bg-transparent text-center no-print p-1 rounded item-input">

                        </button>
                    </td>
                `;
                container.appendChild(tr);
            });
            updateCalculations();
        }

        function numberToWords(num) {
            const a = ['', 'One ', 'Two ', 'Three ', 'Four ', 'Five ', 'Six ', 'Seven ', 'Eight ', 'Nine ', 'Ten ', 'Eleven ', 'Twelve ', 'Thirteen ', 'Fourteen ', 'Fifteen ', 'Sixteen ', 'Seventeen ', 'Eighteen ', 'Nineteen '];
            const b = ['', '', 'Twenty', 'Thirty', 'Forty', 'Fifty', 'Sixty', 'Seventy', 'Eighty', 'Ninety'];

            if ((num = num.toString()).length > 9) return 'Overflow';
            const n = ('000000000' + num).substr(-9).match(/^(\d{2})(\d{2})(\d{2})(\d{1})(\d{2})$/);
            if (!n) return '';
            let str = '';
            str += (n[1] != 0) ? (a[Number(n[1])] || b[n[1][0]] + ' ' + a[n[1][1]]) + 'Crore ' : '';
            str += (n[2] != 0) ? (a[Number(n[2])] || b[n[2][0]] + ' ' + a[n[2][1]]) + 'Lakh ' : '';
            str += (n[3] != 0) ? (a[Number(n[3])] || b[n[3][0]] + ' ' + a[n[3][1]]) + 'Thousand ' : '';
            str += (n[4] != 0) ? (a[Number(n[4])] || b[n[4][0]] + ' ' + a[n[4][1]]) + 'Hundred ' : '';
            str += (n[5] != 0) ? ((str != '') ? 'and ' : '') + (a[Number(n[5])] || b[n[5][0]] + ' ' + a[n[5][1]]) : '';
            return str || 'Zero';
        }

        async function exportAsPNG() {
            showToast("Generating image...");
            const capture = document.getElementById('invoiceCapture');
            const canvas = await html2canvas(capture, {
                scale: 2,
                useCORS: true,
                backgroundColor: "#ffffff",
                windowWidth: 1056
            });
            const link = document.createElement('a');
            link.download = `Invoice_${Date.now()}.png`;
            link.href = canvas.toDataURL('image/png');
            link.click();
            showToast("Invoice Saved Successfully!");
        }

        function resetApp() {

                document.getElementById('custName').value = '';
                document.getElementById('custGstin').value = '';
                document.getElementById('custAddr').value = '';
                syncText('custName', 'displayCustName');
                syncText('custGstin', 'displayCustGstin');
                syncText('custAddr', 'displayCustAddr');
                items = [{ id: Date.now(), desc: '', qty: 1, rate: 0, gst: 18 }];
                renderItems();
            }
        }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            toast.innerText = msg;
