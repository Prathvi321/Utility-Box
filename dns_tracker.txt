    <script>
        const dnsForm = document.getElementById('dnsForm');
        const domainInput = document.getElementById('domainInput');
        const submitBtn = document.getElementById('submitBtn');
        const resultsContainer = document.getElementById('resultsContainer');
        const emptyState = document.getElementById('emptyState');
        const errorState = document.getElementById('errorState');
        const auditTimestamp = document.getElementById('auditTimestamp');

        function switchTab(tabId) {
            // Only hide/show if not printing
            document.querySelectorAll('.tab-view').forEach(v => v.classList.add('hidden'));
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            
            const activeView = document.getElementById(`view-${tabId}`);
            activeView.classList.remove('hidden');
            document.getElementById(`tab-${tabId}`).classList.add('active');
        }

        dnsForm.addEventListener('submit', async (e) => {
            e.preventDefault();
            const domain = domainInput.value.trim().toLowerCase().replace(/^(https?:\/\/)/, '').split('/')[0];
            if (!domain) return;
            
            document.getElementById('displayDomain').textContent = domain;
            auditTimestamp.textContent = new Date().toLocaleString();

            resultsContainer.classList.add('hidden');
            emptyState.classList.add('hidden');
            errorState.classList.add('hidden');
            submitBtn.disabled = true;
            submitBtn.innerHTML = `<div class="loader mr-2"></div> Running...`;

            try {
                await runFullAudit(domain);
                resultsContainer.classList.remove('hidden');
                switchTab('dns'); // Default to DNS tab on load
            } catch (err) {
                errorState.classList.remove('hidden');
                document.getElementById('errorMessage').textContent = err.message;
            } finally {
                submitBtn.disabled = false;
                submitBtn.innerHTML = `<span>Run Audit</span> <i class="fas fa-bolt text-sm"></i>`;
            }
        });

        async function runFullAudit(domain) {
            const recordTypes = ['A', 'AAAA', 'MX', 'NS', 'CNAME', 'TXT'];
            const dnsPromises = recordTypes.map(type => fetchDNS(domain, type));
            const securityPromises = [
                fetchDNS(domain, 'TXT'),
                fetchDNS(`_dmarc.${domain}`, 'TXT')
            ];
            const whoisPromise = fetchWHOIS(domain);
            const headerPromise = fetchHeaders(domain);

            const [dnsResults, securityResults, whoisData, headerData] = await Promise.all([
                Promise.all(dnsPromises),
                Promise.all(securityPromises),
                whoisPromise,
                headerPromise
            ]);

            renderDNS(dnsResults);
            renderSecurity(securityResults);
            renderWHOIS(whoisData);
            renderHealth(dnsResults, securityResults, whoisData);
            renderHeaders(headerData);
        }

        async function fetchDNS(name, type) {
            const res = await fetch(`https://dns.google/resolve?name=${name}&type=${type}`);
            const data = await res.json();
            return { type, name, records: data.Answer || [] };
        }

        async function fetchWHOIS(domain) {
            try {
                const res = await fetch(`https://rdap.org/domain/${domain}`);
                if (!res.ok) return null;
                return await res.json();
            } catch { return null; }
        }

        async function fetchHeaders(domain) {
            try {
                const res = await fetch(`https://api.allorigins.win/get?url=${encodeURIComponent('http://' + domain)}`);
                const data = await res.json();
                return data.contents ? "Analysis Successful: Server is active and responding." : "Limited header access via browser proxy.";
            } catch { return "Headers unreachable due to CORS/Network restrictions."; }
        }

        function renderDNS(results) {
            const grid = document.getElementById('dnsGrid');
            grid.innerHTML = '';
            results.forEach(res => {
                if (res.records.length === 0) return;
                
                const card = document.createElement('div');
                card.className = 'glass-card rounded-2xl p-6';
                card.innerHTML = `
                    <div class="flex items-center justify-between mb-4">
                        <div class="flex items-center gap-2">
                            <span class="record-badge type-${res.type}">${res.type}</span>
                            <h4 class="font-bold text-slate-800">${res.records.length} Record(s)</h4>
                        </div>
                    </div>
                    <div class="space-y-2">
                        ${res.records.map(r => `
                            <div class="code-font text-[11px] bg-slate-50 p-2.5 rounded-lg border border-slate-100 break-all leading-relaxed">
                                ${r.data}
                            </div>
                        `).join('')}
                    </div>
                `;
                grid.appendChild(card);
            });
        }

        function renderSecurity(results) {
            const grid = document.getElementById('securityGrid');
            const [txtRecords, dmarcRecords] = results;
            
            const spf = txtRecords.records.find(r => r.data.includes('v=spf1'));
            const dmarc = dmarcRecords.records.find(r => r.data.includes('v=DMARC1'));

            grid.innerHTML = `
                ${createSecurityCard('SPF (Email Policy)', spf ? 'Configured' : 'Missing', !!spf)}
                ${createSecurityCard('DMARC (Spoof Guard)', dmarc ? 'Enabled' : 'Missing', !!dmarc)}
                ${createSecurityCard('DNSSEC (Data Integrity)', 'Unknown', false)}
            `;
        }

        function createSecurityCard(label, status, isGood) {
            return `
                <div class="p-4 rounded-xl border ${isGood ? 'bg-green-50/50 border-green-100' : 'bg-amber-50/50 border-amber-100'}">
                    <p class="text-[10px] font-black uppercase tracking-widest text-slate-400 mb-1">${label}</p>
                    <div class="flex items-center justify-between">
                        <span class="font-bold text-sm ${isGood ? 'text-green-700' : 'text-amber-700'}">${status}</span>
                        <i class="fas ${isGood ? 'fa-check-circle text-green-500' : 'fa-exclamation-circle text-amber-500'}"></i>
                    </div>
                </div>
            `;
        }

        function renderWHOIS(data) {
            const content = document.getElementById('whoisContent');
            if (!data) {
                content.innerHTML = `<div class="p-8 text-center text-slate-400 italic bg-slate-50 rounded-xl border border-dashed">No registration data available via RDAP protocol.</div>`;
                return;
            }

            const events = data.events || [];
            const registrar = data.entities ? data.entities[0]?.vcardArray?.[1]?.find(i => i[0] === 'fn')?.[3] : 'N/A';

            content.innerHTML = `
                <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
                    <div class="p-4 bg-slate-50 rounded-xl border border-slate-100">
                        <p class="text-[10px] text-slate-400 font-bold mb-1 uppercase tracking-widest">Registrar</p>
                        <p class="font-bold text-slate-800 text-sm truncate">${registrar}</p>
                    </div>
                    <div class="p-4 bg-slate-50 rounded-xl border border-slate-100">
                        <p class="text-[10px] text-slate-400 font-bold mb-1 uppercase tracking-widest">Domain Status</p>
                        <p class="font-bold text-blue-600 text-sm">${(data.status || ['Active'])[0]}</p>
                    </div>
                    ${events.slice(0, 4).map(e => `
                        <div class="p-4 bg-slate-50 rounded-xl border border-slate-100">
                            <p class="text-[10px] text-slate-400 font-bold mb-1 uppercase tracking-widest">${e.eventAction.replace(/([A-Z])/g, ' $1')}</p>
                            <p class="font-bold text-slate-800 text-sm">${new Date(e.eventDate).toLocaleDateString(undefined, {year: 'numeric', month: 'long', day: 'numeric'})}</p>
                        </div>
                    `).join('')}
                </div>
            `;
        }

        function renderHealth(dns, security, whois) {
            const container = document.getElementById('healthChecks');
            const checks = [];

            const hasA = dns.find(r => r.type === 'A')?.records.length > 0;
            checks.push({ label: 'A Record Presence', pass: hasA, desc: hasA ? 'Domain resolves to a valid IPv4 address.' : 'Missing A record. The website will not load.' });

            const hasMX = dns.find(r => r.type === 'MX')?.records.length > 0;
            checks.push({ label: 'MX Record Configuration', pass: hasMX, desc: hasMX ? 'Mail exchange servers are correctly defined.' : 'No MX records found. This domain cannot receive emails.' });

            const hasSPF = security[0].records.some(r => r.data.includes('v=spf1'));
            checks.push({ label: 'SPF Security Record', pass: hasSPF, desc: hasSPF ? 'SPF policy exists to prevent unauthorized email sending.' : 'Missing SPF record. High risk of email spoofing.' });

            const hasNS = dns.find(r => r.type === 'NS')?.records.length > 0;
            checks.push({ label: 'Name Server Redundancy', pass: hasNS, desc: hasNS ? 'Authoritative name servers are responding.' : 'Critical error: No name servers detected.' });

            container.innerHTML = checks.map(c => `
                <div class="flex items-start gap-4 p-4 rounded-xl border ${c.pass ? 'bg-white border-slate-100' : 'bg-red-50 border-red-100'}">
                    <div class="mt-0.5">
                        <i class="fas ${c.pass ? 'fa-check-circle text-green-500' : 'fa-times-circle text-red-400'} text-lg"></i>
                    </div>
                    <div>
                        <h5 class="font-bold text-sm text-slate-800">${c.label}</h5>
                        <p class="text-xs text-slate-500 mt-1">${c.desc}</p>
                    </div>
                </div>
            `).join('');
        }

        function renderHeaders(data) {
            const pre = document.getElementById('headersContent');
            pre.textContent = "Report Summary:\n" + data + 
                           "\n\nNote: Detailed HTTP response headers (like Server, X-Frame-Options) are blocked by browser CORS security policies. For a full security header audit, a server-side scanner is recommended.";
        }
    </script>
